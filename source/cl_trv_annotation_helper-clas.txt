CLASS CL_TRV_ANNOTATION_HELPER IMPLEMENTATION.


  METHOD add_default_filter_value.
    co_ann_target  = io_vocan_model->create_annotations_target( iv_target = iv_namespace  && '.' && iv_entity_type && '/' && iv_property )   ##NO_TEXT.
    IF iv_bool_value IS SUPPLIED.
      co_ann_target->create_annotation( iv_term = 'com.sap.vocabularies.Common.v1.FilterDefaultValue' )->create_simple_value( )->set_boolean( iv_value = iv_bool_value ).
    ENDIF.
    IF iv_string_value IS SUPPLIED.
      co_ann_target->create_annotation( iv_term = 'com.sap.vocabularies.Common.v1.FilterDefaultValue' )->create_simple_value( )->set_string( iv_value = iv_string_value ).
    ENDIF.

  ENDMETHOD.


 METHOD add_display_parameter.
   DATA:
     lo_record TYPE REF TO /iwbep/if_mgw_vocan_record.

   lo_record = mo_parameters->create_record( 'com.sap.vocabularies.Common.v1.ValueListParameterDisplayOnly' ).
   lo_record->create_property( 'ValueListProperty' )->create_simple_value( )->set_string( iv_valuelist_property ).

   ro_annotation = me.
 ENDMETHOD.


  METHOD add_field_controls.

    DATA: lt_properties       TYPE /iwbep/if_mgw_med_odata_types=>ty_t_mgw_odata_properties,
          ls_property         TYPE /iwbep/if_mgw_med_odata_types=>ty_s_mgw_odata_property,
          lv_property         TYPE /iwbep/if_mgw_med_odata_types=>ty_e_med_entity_name,
          ls_data_fc_property TYPE LINE OF tt_data_fc_properties,
          lv_fc_property      TYPE /iwbep/med_annotation_value.


    lt_properties = io_model->get_entity_type( iv_entity_type )->get_properties( ).

    LOOP AT lt_properties INTO ls_property.
      DATA: len type i,
            propertyname type string.
      propertyname = ls_property-external_name.
      translate propertyname to upper case.
      len = strlen( propertyname ) - 2.
      if len >= 0 and propertyname+len(2) = 'FC'.
        lv_property = ls_property-external_name(len).
        lv_fc_property = ls_property-external_name.
        set_field_control(
          EXPORTING
            io_model       =   io_model
            iv_entity_type =  iv_entity_type
            iv_property    =  lv_property
            iv_fc_property =  lv_fc_property
        ).
      ENDIF.
    ENDLOOP.

    LOOP AT tt_data_fc_properties INTO ls_data_fc_property. "Additionaly properties to map
      lv_property = ls_data_fc_property-data_property.
      lv_fc_property = ls_data_fc_property-fc_property.
      set_field_control(
        EXPORTING
          io_model       =   io_model
          iv_entity_type =  iv_entity_type
          iv_property    =  lv_property
          iv_fc_property =  lv_fc_property
      ).
    ENDLOOP.

  ENDMETHOD.

    lo_record->create_property( 'ValueListProperty' )->create_simple_value( )->set_string( iv_valuelist_property ).

    ro_annotation = me.
  ENDMETHOD.
